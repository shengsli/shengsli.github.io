static int find (int list[], int n, int key) // 1
{
	int lo = 0, mid; // 2
	int hi = n - 1;  // 3
	int result = -1; // 4
	while ((hi >= lo) && (result == -1)) { // 5
		mid = (lo + hi) / 2;  // 6
		if (list[mid] == key) // 7
			result = mid;     //8
		else if (list[mid] > key) // 9
			hi = mid - 1; // 10
		else
			lo = mid + 1; // 11
	}
	return result; // 12
}
